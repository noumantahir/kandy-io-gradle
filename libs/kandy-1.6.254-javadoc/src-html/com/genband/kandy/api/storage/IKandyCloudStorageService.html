010package com.genband.kandy.api.storage;
011
012
013import com.genband.kandy.api.services.chats.IKandyFileItem;
014import com.genband.kandy.api.services.chats.KandyThumbnailSize;
015import com.genband.kandy.api.services.common.KandyResponseCancelListener;
016import com.genband.kandy.api.services.common.KandyResponseListener;
017import com.genband.kandy.api.services.common.KandyResponseProgressListener;
018import com.genband.kandy.api.services.common.KandyUploadProgressListener;
019
020/**
021 * <b>Kandy Cloud Storage </b> is a secure online storage.
022 * <p>
023 * It lets you access your photos/videos/files from anywhere.<br>
024 * You can use {@link #KandyCloudStorageService()} to store and retrieve your
025 * data at any time, from any device to the cloud.<br>
026 * You can accomplish these tasks using the simple and intuitive API.
027 * 
028 * @author Fabio
029 *
030 */
031public interface IKandyCloudStorageService {
032        
033        /**
034         * This method add the file to your could storage.<br>
035         * The file can be accessed using {@link #downloadMedia(IKandyFileItem, KandyResponseListener)} API. 
036         * @param fileItem The file to be uploaded
037         * @param listener The callback that will be invoked 
038         */
039        public void uploadMedia(IKandyFileItem kandyFileItem, final KandyUploadProgressListener listener);
040        
041        /**
042         * This method let you retrieve a file, that was previously uploaded using {@link #uploadMedia(IKandyFileItem, KandyResponseListener)} to the cloud. 
043         * @param file The file to retrieve
044         * @param listener The callback that will be invoked
045         */
046        public void downloadMedia(IKandyFileItem kandyFileItem, KandyResponseProgressListener listener);
047        
048        /**
049         * This method let you retrieve a file thumbnail, that was previously uploaded using {@link #uploadKandyFile(IKandyFileItem, KandyResponseListener)} to the cloud. 
050         * @param file The file to retrieve
051         * @param thumbnailSize size of the thumbnail to download
052         * @param listener The callback that will be invoked
053         */
054        public void downloadMediaThumbnail (IKandyFileItem kandyFileItem, KandyThumbnailSize thumbnailSize, KandyResponseProgressListener listener);
055        
056        /**
057         * This method used for cancel the media transfer(upload/download)
058         * @param message which transfer must be canceled
059         * @param listener The callback that will be invoked with result of operation
060         */
061        public void cancelMediaTransfer(IKandyFileItem kandyFileItem, KandyResponseCancelListener listener);
062        
063}
