010package com.genband.kandy.api.access;
011
012import com.genband.kandy.api.services.calls.KandyRecord;
013
014
015/**
016 * This interface represents the methods of a registrationService.
017 * <p>
018 * Registration service are used for:
019 * <ul>
020 * <li>registering/unregistering to/from the server
021 * <li>getting registration state, and for getting already registered services
022 * </ul>
023Success and failure callback are handled by the RegistrationListener interface.
024 */
025public interface IKandyAccess
026{
027        /** register to ongoing notification */
028        public void registerNotificationListener(KandyConnectServiceNotificationListener listener);
029
030        /** unregister from ongoing notification */
031        public void unregisterNotificationListener(KandyConnectServiceNotificationListener listener);
032
033        /**
034         * This method connect user to server
035         * 
036         * @param user
037         *            The username value that are used in registration process.
038         * @param password
039         *            The password value that are used in registration process
040         * @param listener The callback that will be invoked
041         */
042        public void login(KandyRecord user, String password, KandyLoginResponseListener listener);
043        
044        /**
045         * This method connect user to server with user token
046         * 
047         * @param userAccessToken
048         *            The use access token string
049         * @param listener
050         *            The callback that will be invoked
051         */
052        public void login(String userAccessToken, KandyLoginResponseListener listener);
053        
054        /**
055         * This method disconnect user from the server
056         * 
057         * @param listener The callback that will be invoked
058         */
059        public void logout(KandyLogoutResponseListener listener);
060
061        /**
062         * @return the registration process state
063         */
064        public KandyRegistrationState getRegistrationState();
065
066        /**
067         * @return the connection process state
068         */
069        public KandyConnectionState getConnectionState();
070
071        /**
072         * Recommended to use when application moves to the background. 
073         * The call initialization will be longer.
074         */
075        public void goToSleep();
076        
077}
