010package com.genband.kandy.api.services.push;
011
012import android.content.Context;
013import android.content.Intent;
014
015import com.genband.kandy.api.services.common.KandyResponseListener;
016import com.genband.kandy.api.services.events.KandyEventsServiceNotificationListener;
017import com.genband.kandy.providers.kandy.services.KandyEventsCoreService;
018/**
019 * This interface is used to handle push credentials, events and notifications.
020 * <p>Push notifications let your application receive events, even when the user is not actively 
021 * using your application. On Android devices, when a device receives a push notification, your 
022 * application's icon and a message appear in the status bar. When the user taps the notification, 
023 * they are sent to your application.
024 * <p>The Kandy SDK provides push notifications using Google Cloud Messaging (GCM) if possible. 
025 * On devices that do not support GCM, Kandy will use a background service that maintains a 
026 * persistent connection to the Kandy Cloud to deliver pushes.
027 * <p><b><i>Configure</b></i>
028 * <ul>
029 * <li>Receivers & Services
030 * <p>To use push notifications, your application must register the following service and 
031 * broadcast receivers. Add the following XML to your AndroidManifest.xml file immediately 
032 * before the closing <tt>&lt;/application&gt;</tt> tag:
033 * <pre><code>
034 * &lt;receiver
035 *      android:name="com.genband.kandy.api.services.push.KandyGCMPushReceiver"
036 *      android:permission="com.google.android.c2dm.permission.SEND" &gt;
037 *      &lt;intent-filter&gt;
038 *      &lt;action android:name="com.google.android.c2dm.intent.RECEIVE" /&gt;
039 *      &lt;action android:name="com.google.android.c2dm.intent.REGISTRATION" /&gt;
040 *      &lt;category android:name="com.kandy.sampleapp" /&gt;
041 *      &lt;/intent-filter&gt;
042 * &lt;/receiver&gt;
043 * &lt;service
044 *      android:name="com.genband.kandy.api.services.push.KandyGCMPushService"
045 *      android:enabled="true" /&gt;
046 * </code></pre>
047 * <p>If you modify the package name of the tutorial app, change the android:name attribute of 
048 * <tt>&lt;category&gt;</tt> element above to match your application's package name. If you choose to subclass 
049 * KandyGCMPushReceiver, change com.genband.kandy.services.push.model.KandyGCMPushReceiver above 
050 * to your class name.
051 * <li>Permissions
052 * <p>Your application must also have the permissions needed to receive pushes and show notifications. 
053 * Make sure that these permissions are present in your AndroidManifest.xml file, typically immediately 
054 * before the <tt>&lt;application&gt;</tt> tag:
055 * <pre><code>
056 * &lt;permission
057 *      android:name="com.kandy.sampleapp.permission.C2D_MESSAGE"
058 *      android:protectionLevel="signature" /&gt;
059 * &lt;uses-permission android:name="com.kandy.sampleapp.permission.C2D_MESSAGE" /&gt;
060 * &lt;uses-permission android:name="com.google.android.c2dm.permission.RECEIVE" /&gt;
061 * </code></pre>
062 * <p>If you modify the package name of the tutorial app, change the android:name attribute in the last 
063 * two lines of the snippet above to match your application's package name.
064 * <li>Notification icon
065 * </ul>
066 * <p><b><i>Enable</b></i>
067 * <p>Your device needs to register itself for push notifications.
068 * <p>To enable your device for push:
069 * <pre><code>
070 * Kandy.getServices().getPushService().enablePushNotification(new KandyResponseListener() {
071 *      
072 *      <t><b>@Override</b></t>
073 *      public void onRequestFailed(int responseCode, String err) {
074 *              //TODO insert your code here
075 *      }
076 * 
077 *      <t><b>@Override</b></t>
078 *      public void onRequestSucceded() {
079 *              //TODO insert your code here
080 *      }
081 * });
082 * </code></pre>
083*/
084
085
086public interface IKandyPushService
087{
088        
089        /**
090         * Start receiving push notifications.
091         * <p>
092         * The received events will be triggered thru {@link KandyEventsCoreService}
093         * . make sure the register a listener with
094         * {@link KandyEventsCoreService#registerServiceNotificationsListener(KandyEventsServiceNotificationListener)}
095         * <br><b>Note:</b><br>Must be called after successfully login.
096         * @param registrationId token for the push notification must be passed to Kandy for registartion on Kandy Push Notifications
097         * @param listener Listener which handles the response of the operation
098         * 
099         */
100        public void enablePushNotification(String registrationId, KandyResponseListener listener);
101
102        /**
103         * Handle the Kandy push notification
104         * @param context
105         * @param intent with Kandy Push Notification data to be handled
106         * @param listener - {@link KandyResponseListener} which handles the response of the operationListener which handles the response of the operation
107         */
108        public void handlePushNotification(Context context, Intent intent, KandyResponseListener listener);
109        
110        /**
111         * Stop receiving push events
112         */
113        public void disablePushNotification(KandyResponseListener listener);
114        
115        
116}
