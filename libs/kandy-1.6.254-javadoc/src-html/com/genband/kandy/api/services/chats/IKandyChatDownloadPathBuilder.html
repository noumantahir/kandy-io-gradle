017        
018        /**
019         *  This methods builds and returns the absolute path for downloaded files
020         *
021         *  @param sender       {@link KandyRecord} Sender record
022         *  @param recipient    {@link KandyRecord} Recipient record
023         *  @param fileItem     The downloaded file item
024         *  @param isThumbnail  BOOL indicating if the file item downloaded is a thumbnail
025         *
026         *  @return The path to store downloaded, received media content.
027         */

030        /**

036}
