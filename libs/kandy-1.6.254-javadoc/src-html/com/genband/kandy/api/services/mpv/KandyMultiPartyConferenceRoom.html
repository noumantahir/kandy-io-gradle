001package com.genband.kandy.api.services.mpv;
002
003import com.genband.kandy.api.utils.KandyLog;
004import com.genband.kandy.config.JSONKeys.JSONMultiPartiConferenceRoom;
005import com.genband.kandy.core.common.IJSONParse;
006
007import org.json.JSONException;
008import org.json.JSONObject;
009
010public class KandyMultiPartyConferenceRoom implements IJSONParse
011{
012        private static final String TAG = "KandyMultiPartyConferenceRoom";
013        private String mRoomNumber;
014        private String mRoomPinCode;
015        private String mRoomPSTNNumber;
016        private String mConferenceId;
017        
018        
019        public KandyMultiPartyConferenceRoom()
020        {
021                
022        }
023
024        /**
025         * @return the mRoomId
026         */
027        public String getConferenceId()
028        {
029                return mConferenceId;
030        }
031
032
033        public String getRoomNumber()
034        {
035                return mRoomNumber;
036        }
037
038        public String getRoomPinCode()
039        {
040                return mRoomPinCode;
041        }
042
043        /**
044         * @return the mRoomPSTNNumber
045         */
046        public String getRoomPSTNNumber()
047        {
048                return mRoomPSTNNumber;
049        }
050
051        
052        
053        @Override
054        public String toString() {
055                return "KandyMultiPartyConferenceRoom [mRoomNumber=" + mRoomNumber + 
056                                ", mRoomPinCode=" + mRoomPinCode + 
057                                ", mRoomPSTNNumber=" + mRoomPSTNNumber + 
058                                ", mConferenceId=" + mConferenceId + "]";
059        }
060
061        public void initFromString(String jsonString)
062        {
063                JSONObject json;
064                try {
065                        json = new JSONObject(jsonString);
066                        initFromJson(json);
067                } 
068                catch (JSONException e) 
069                {
070                        KandyLog.w(TAG, "initFromJson: " + " " + e.getLocalizedMessage());
071                }
072        }
073        
074        @Override
075        public void initFromJson(JSONObject jsonObject)
076        {
077                
078                        try
079                        {
080                                JSONObject confDetails = jsonObject.getJSONObject(JSONMultiPartiConferenceRoom.KEY_ROOM_DETAILS);
081                                if(confDetails != null)
082                                {
083                                        try
084                                        {
085                                        mConferenceId = confDetails.getString(JSONMultiPartiConferenceRoom.KEY_ROOM_ID);
086                                        }
087                                        catch (Exception e) 
088                                        {
089                                                KandyLog.w(TAG, "initFromJson: " + " " + e.getLocalizedMessage());
090                                        }
091                                        try
092                                        {
093                                        mRoomNumber = confDetails.getString(JSONMultiPartiConferenceRoom.KEY_ROOM_IP_ACCESS);
094                                        }
095                                        catch (Exception e) 
096                                        {
097                                                KandyLog.w(TAG, "initFromJson: " + " " + e.getLocalizedMessage());
098                                        }
099                                        try
100                                        {
101                                        mRoomPSTNNumber = confDetails.getString(JSONMultiPartiConferenceRoom.KEY_ROOM_PSTN_ACCESS);
102                                        }
103                                        catch (Exception e) 
104                                        {
105                                                KandyLog.w(TAG, "initFromJson: " + " " + e.getLocalizedMessage());
106                                        }
107                                        try
108                                        {
109                                        mRoomPinCode = confDetails.getString(JSONMultiPartiConferenceRoom.KEY_ROOM_PIN_CODE);
110                                        }
111                                        catch (Exception e) 
112                                        {
113                                                KandyLog.w(TAG, "initFromJson: " + " " + e.getLocalizedMessage());
114                                        }
115                                }
116                        }
117                        catch (Exception e) 
118                        {
119                                KandyLog.w(TAG, "initFromJson: " + " " + e.getLocalizedMessage());
120                        }
121                
122        }
123
124        @Override
125        public JSONObject toJson()
126        {
127                JSONObject roomJsonObjectWrapper = new JSONObject();
128                JSONObject roomJsonObject = new JSONObject();
129                try
130                        {
131                                roomJsonObject.put(JSONMultiPartiConferenceRoom.KEY_ROOM_ID,mConferenceId);
132                        }
133                        catch (JSONException e)
134                        {
135                                KandyLog.w(TAG, "toJson: " + " " + e.getLocalizedMessage());
136                        }
137                try
138                {
139                        roomJsonObject.put(JSONMultiPartiConferenceRoom.KEY_ROOM_IP_ACCESS, mRoomNumber);
140                }
141                catch (JSONException e)
142                {
143                        KandyLog.w(TAG, "toJson: " + " " + e.getLocalizedMessage());
144                }
145                try
146                {
147                        roomJsonObject.put(JSONMultiPartiConferenceRoom.KEY_ROOM_PSTN_ACCESS, mRoomPSTNNumber);
148                }
149                catch (JSONException e)
150                {
151                        KandyLog.w(TAG, "toJson: " + " " + e.getLocalizedMessage());
152                }
153                try
154                {
155                        roomJsonObject.put(JSONMultiPartiConferenceRoom.KEY_ROOM_PIN_CODE, mRoomPinCode);
156                }
157                catch (JSONException e)
158                {
159                        KandyLog.w(TAG, "toJson: " + " " + e.getLocalizedMessage());
160                }
161                try
162                {
163                        roomJsonObjectWrapper.put(JSONMultiPartiConferenceRoom.KEY_ROOM_DETAILS, roomJsonObject);
164                }
165                catch (JSONException e)
166                {
167                        KandyLog.w(TAG, "toJson: " + " " + e.getLocalizedMessage());
168                }
169                
170                return roomJsonObjectWrapper;
171        }
172
173        
174}
