001package com.genband.kandy.api.services.mpv;
002
003import java.util.List;
004
005import com.genband.kandy.api.services.common.KandyResponseListener;
006
007public interface IKandyMultiPartyConferenceService
008{
009        
010        /**
011         * Create Room and invite users to it
012         * @param {@link KandyMultiPartyConferenceInvitees}} invitees list of contacts to invite
013         * @param {@link KandyMultiPartyConferenceCreateAndInviteListener} listener
014         * @param annotation
015         */
016        public void createRoomAndInvite(KandyMultiPartyConferenceAnnotation annotation, KandyMultiPartyConferenceInvitees invitees, KandyMultiPartyConferenceCreateAndInviteListener listener);
017        
018        /**
019         * Invite users to the room
020         * @param conferenceId
021         * @param invitees {@link KandyMultiPartyConferenceInvitees}} list of contacts to invite
022         * @param listener {@link KandyResponseListener}
023         */
024        void invite(String conferenceId, KandyMultiPartyConferenceInvitees invitees,KandyMultiPartyConferenceInviteListener listener);
025        /**
026         * create multiparty conference room
027         * @param annotation
028         * @param litener
029         */
030        public void createRoom(KandyMultiPartyConferenceAnnotation annotation, KandyMultiPartyConferenceCreateListener litener);
031        
032        /**
033         * destroy existing room
034         * @param conferenceId mandatory
035         * @param listener {@link KandyResponseListener}
036         */
037        public void destroyRoom(String conferenceId, KandyResponseListener listener);
038        
039        /**
040         * join existing room
041         * @param conferenceId mandatory field
042         * @param nickName send server a nickname you want to be displayed with
043         * @param listener {@link KandyMultiPartyConferenceCreateAndInviteListener} 
044         */
045        public void join(String conferenceId, String nickName, KandyResponseListener listener);
046        
047        /**
048         * 
049         * @param conferenceId
050         * @param {@link KandyResponseListener} listener
051         */
052        public void leave(String conferenceId, KandyResponseListener listener);
053        
054        /**
055         * get conference room details, call duration, participants 
056         * @param conferenceId mandatory
057         * @param {@link KandyMultiPartyConferenceRoomDetailsListener} listener
058         */
059        public void getRoomDetails(String conferenceId, KandyMultiPartyConferenceRoomDetailsListener listener);
060        
061        /**
062         * update participant nickname 
063         * @param conferenceId mandatory
064         * @param participantId mandatory
065         * @param nickname - the new nickName you want to be update
066         * @param {@link KandyResponseListener} listener
067         */
068        public void updateParticipantName(String conferenceId, String participantId, String nickname,  KandyResponseListener listener);
069        
070        /**
071         * Update participant actions in room - action can be "mute" ,"unmute", "enableVideo", "disableVideo" , "hold", "unhold" ,"remove"
072         * @param conferenceId mandatory
073         * @param {@link KandyMultiPartyConferenceParticipantActionParmas} participantActionParams - pair of participant id and the action you want to set to this participant
074         * @param {@link KandyMultiPartyConferenceUpdateParticipantActionsListener} listener
075         */
076        public void updateParticipantActions(String conferenceId, List<KandyMultiPartyConferenceParticipantActionParmas> participantActionParams, KandyMultiPartyConferenceUpdateParticipantActionsListener listener);
077        
078        /**
079         * Register to incoming events the client receives from the group service
080         * @param listener to be registered/added to group service notifications
081         */
082        public void registerNotificationListener(IKandyMultiPartyConferenceNotificationListener listener);
083
084        /**
085         * Stop listening to group notifications
086         * @param listener to be unregistered/removed from group service notifications
087         */
088        public void unregisterNotificationListener(IKandyMultiPartyConferenceNotificationListener listener);
089
090        
091}
