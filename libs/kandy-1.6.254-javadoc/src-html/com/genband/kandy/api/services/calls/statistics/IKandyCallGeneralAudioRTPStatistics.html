010package com.genband.kandy.api.services.calls.statistics;
011
012/**
013 * Defines general statistics of an audio call.
014 *  (We have 3 types of audio/video call statistics : "receive" (which data is received); "sent" (which data was sent) and "general")
015 */
016public interface IKandyCallGeneralAudioRTPStatistics extends IKandyCallStatistic {
017        /**
018         * // host/stun/relay candidates
019         */
020        public String getLocalCandidateType();
021
022        /**
023         * // host/stun/relay candidates
024         */
025        public String getRemoteCandidateType();
026
027        /**
028         * local address (incoming media IP)
029         */
030        public String getLocalAddress();
031
032        /**
033         * remote address (remote media IP)
034         */
035        public String getRemoteAddress();
036
037        /**
038         * packets sent
039         */
040        public int getPacketsSent();
041
042        /**
043         * cumulative amount of bytes (audio + video)
044         */
045        public long getBytesReceived();
046
047        /**
048         * cumulative amount of bytes (audio + video)
049         */
050        public long getBytesSent();
051}
