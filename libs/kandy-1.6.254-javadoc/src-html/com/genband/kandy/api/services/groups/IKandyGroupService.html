010package com.genband.kandy.api.services.groups;
011
012import android.net.Uri;
013
014import com.genband.kandy.api.services.calls.KandyRecord;
015import com.genband.kandy.api.services.chats.KandyThumbnailSize;
016import com.genband.kandy.api.services.common.KandyResponseListener;
017import com.genband.kandy.api.services.common.KandyResponseProgressListener;
018
019import java.util.List;
020
021/**
022 * Interface represents the actions for group and group's participants management
023 *
024 */
025public interface IKandyGroupService {
026        
027        /**
028         * Register to incoming events the client receives from the group service
029         * @param listener to be registered/added to group service notifications
030         */
031        public void registerNotificationListener(KandyGroupServiceNotificationListener listener);
032
033        /**
034         * Stop listening to group notifications
035         * @param listener to be unregistered/removed from group service notifications
036         */
037        public void unregisterNotificationListener(KandyGroupServiceNotificationListener listener);
038
039        /**
040         * Create a KandyGroup by the received KandyGroupParams
041         * @param groupParams {@link KandyGroupParams} to use for creating the KandyGroup
042         * @param responseListener will be called upon completion, will contain error in case of failure or {@link KandyGroup} in case of success
043         * see {@link KandyGroupResponseListener}
044         */
045        public void createGroup(KandyGroupParams groupParams, KandyGroupResponseListener responseListener);
046
047        /**
048         * Destroy group by group ID
049         * @param groupId the ID of the group to destroy <b>NOT NULL</b>
050         * @param responseListener {@link KandyResponseListener} calls the success callback or fail with error callback 
051         */
052        public void destroyGroup(KandyRecord groupId, KandyResponseListener responseListener);
053
054        /**
055         * Update a KandyGroup by the received {@link KandyGroupParams}
056         * @param groupId the ID of the group to update
057         * @param groupParams {@link KandyGroupParams} to use for updating the {@link KandyGroup}
058         * @param responseListener {@link KandyGroupResponseListener} will be called upon completion, will contain error in case of failure or KandyGroup in case of success
059         * on success will be returned the {@link KandyGroup} instance
060         * 
061         */
062        public void updateGroup(KandyRecord groupId, KandyGroupParams groupParams, KandyGroupResponseListener responseListener);
063
064        /**
065         * Update the group name
066         * @param groupId the ID of the group to update
067         * @param groupName new name to update
068         * @param responseListener {@link KandyGroupResponseListener}  will be called upon completion, will contain error in case of failure or KandyGroup in case of success
069         * on success will be returned the {@link KandyGroup} instance
070         */
071        public void updateGroupName(KandyRecord groupId, String groupName, KandyGroupResponseListener responseListener);
072
073        /**
074         * Update the group image
075         * @param groupId the ID of the group to update
076         * @param groupImageUri new image to be updated to 
077         * @param responseListener {@link KandyGroupResponseListener} will be called upon completion, will contain error in case of failure or KandyGroup in case of success
078         */
079        public void updateGroupImage(KandyRecord groupId, Uri groupImageUri, KandyGroupResponseListener responseListener);
080
081        /**
082         * Add participants to the group
083         * @param groupId the ID of the group
084         * @param participants list of  {@link KandyRecord}s of participants to add
085         * @param responseListener {@link KandyGroupResponseListener} will be called upon completion, will contain error in case of failure or KandyGroup in case of success
086         */
087        public void addParticipants(KandyRecord groupId, List<KandyRecord> participants, KandyGroupResponseListener responseListener);
088
089        /**
090         * Remove participants from the group
091         * @param groupId the ID of the group
092         * @param participants  list of  {@link KandyRecord}s of participants to remove
093         * @param responseListener  {@link KandyGroupResponseListener} will be called upon completion, will contain error in case of failure or KandyGroup in case of success
094         */
095        public void removeParticipants(KandyRecord groupId, List<KandyRecord> participants, KandyGroupResponseListener responseListener);
096
097        /**
098         * Mute group - push notifications from this group will not be received
099         * @param groupId groupId the ID of the group
100         * @param responseListener  {@link KandyGroupResponseListener} will be called upon completion, will contain error in case of failure or KandyGroup in case of success
101         *  on success will be returned  {@link KandyGroup} instance
102         */
103        public void muteGroup(KandyRecord groupId, KandyGroupResponseListener responseListener);
104
105        /**
106         * Unmute group - push notifications from this group will be received
107         * @param groupId groupId the ID of the group
108         * @param responseListener  {@link KandyGroupResponseListener} will be called upon completion, will contain error in case of failure or KandyGroup in case of success
109         * on success will be returned  {@link KandyGroup} instance
110         */
111        public void unmuteGroup(KandyRecord groupId, KandyGroupResponseListener responseListener);
112
113        /**
114         * Mute participants - push notifications from this participants will not be received
115         * @param groupId group where the selected members will be muted
116         * @param participants list of participants to be muted
117         * @param responseListener  {@link KandyGroupResponseListener} calls the success callback or fail with error callback
118         * on success will be returned  {@link KandyGroup} instance
119         */
120        public void muteParticipants(KandyRecord groupId, List<KandyRecord> participants, KandyGroupResponseListener responseListener);
121
122        /**
123         * Unmute participants - push notifications from this participants will be received
124         * @param groupId the ID of the group
125         * @param participants list of the participants to mute
126         * @param responseListener  {@link KandyGroupResponseListener} will be called upon completion, will contain error
127         *  in case of failure or KandyGroup in case of success
128         * on success will be returned  {@link KandyGroup} instance
129         */
130        public void unmuteParticipants(KandyRecord groupId, List<KandyRecord> participants, KandyGroupResponseListener responseListener);
131
132        
133        /**
134         * Get group details by group id
135         * @param groupId   the ID of the group to get the details of
136         * @param responseListener {@link KandyGroupResponseListener} will be called upon completion, will contain error
137         *  in case of failure or KandyGroup in case of success
138         * on success will be returned  {@link KandyGroup} instance with group details
139         */
140        public void getGroupById(KandyRecord groupId, KandyGroupResponseListener responseListener);
141
142        /**
143         * Get all my groups
144         * @param responseListener {@link KandyGroupsResponseListener} will be called upon completion, will contain error
145         *  in case of failure or KandyGroup in case of success
146         */
147        public void getMyGroups(KandyGroupsResponseListener responseListener);
148
149        /**
150         * Leave a group
151         * @param groupId the ID of the group to leave
152         * @param responseListener {@link KandyResponseListener} will be called upon completion, will contain error incase of failure
153         */
154        public void leaveGroup(KandyRecord groupId, KandyResponseListener responseListener);
155        
156        /**
157         * Download the group image
158         * @param groupId id of the group, which image to download
159         * @param progressListener track the downloa dprogress and notify when download succeeded with file's uri on device or failed with error
160         */
161        public void downloadGroupImage(KandyRecord groupId, final KandyResponseProgressListener progressListener);
162        
163        /**
164         * Download the goup's image thumnail
165         * @param groupId  groupId id of the group, which thumbnail to download
166         * @param thumbnailSize size of the thumbnail to be downloaded {@link KandyThumbnailSize}
167         * @param progressListener track the downloa dprogress and notify when download succeeded with file's uri on device or failed with error
168         */
169        public void downloadGroupImageThumbnail(KandyRecord groupId, KandyThumbnailSize thumbnailSize, final KandyResponseProgressListener progressListener);
170        
171        /**
172         * Remove the Group image(avatar)
173         * @param groupId group wich image want to remove
174         * @param responseListener {@link KandyResponseListener} calls the success callback or fail with error callback
175         */
176        public void removeGroupImage(KandyRecord groupId, KandyGroupResponseListener responseListener);
177        
178        /**
179         * Gets the settings object to configure service specific configurations
180         * @return The settings instance
181         */
182        public KandyGroupSettings getSettings();
183
184}
