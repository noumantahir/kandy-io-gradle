010package com.genband.kandy.api.services.addressbook;
011
012import com.genband.kandy.api.services.common.KandyResponseListener;
013import com.genband.kandy.api.utils.KandyLog;
014import com.genband.kandy.core.KandyCore;
015
016public class KandyAddressBookService implements IKandyAddressBookService
017{
018        private static final String TAG = "KandyAddressBookService";
019
020        private static KandyAddressBookService sInstance = null;
021
022        private KandyAddressBookService()
023        {
024                KandyLog.d(TAG, "initialize " + TAG);
025        }
026
027        public static synchronized KandyAddressBookService getInstance()
028        {
029                if (sInstance == null)
030                {
031                        sInstance = new KandyAddressBookService();
032                }
033                return sInstance;
034        }
035        
036        @Override
037        public void getDeviceContacts(KandyDeviceContactsFilter[] filters, KandyContactsListener listner) {
038                
039                KandyCore.getInstance().getRouting().getAddressBookService().getDeviceContacts(filters, listner);
040                
041        }
042
043        @Override
044        public void getDomainDirectoryContacts(KandyContactsListener listener)
045        {
046                KandyCore.getInstance().getRouting().getAddressBookService().getDomainDirectoryContacts(listener);              
047        }
048        
049        @Override
050        public void getFilteredDomainDirectoryContacts(KandyDomainContactFilter filter, boolean isCaseSensitive,  String search_string, KandyContactsListener listener) {
051                KandyCore.getInstance().getRouting().getAddressBookService().getFilteredDomainDirectoryContacts(filter, isCaseSensitive, search_string, listener);
052        }
053
054        @Override
055        public void registerNotificationListener(KandyAddressBookServiceNotificationListener listener)
056        {
057                KandyCore.getInstance().getRouting().getAddressBookService().registerNotificationListener(listener);
058        }
059
060        @Override
061        public void unregisterNotificationListener(KandyAddressBookServiceNotificationListener listener)
062        {
063                KandyCore.getInstance().getRouting().getAddressBookService().unregisterNotificationListener(listener);          
064        }
065
066        @Override
067        public void getPersonalAddressBook(KandyContactsListener listener) 
068        {
069                KandyCore.getInstance().getRouting().getAddressBookService().getPersonalAddressBook(listener);
070        }
071
072    @Override
073        public void addContactToPersonalAddressBook(KandyContactParams contactParams, KandyContactListener listener)
074        {
075                KandyCore.getInstance().getRouting().getAddressBookService().addContactToPersonalAddressBook(contactParams, listener);
076        }
077
078        @Override
079        public void removePersonalAddressBookContact(String contactId, KandyResponseListener listener)
080        {
081                KandyCore.getInstance().getRouting().getAddressBookService().removePersonalAddressBookContact(contactId, listener);
082        }
083}
