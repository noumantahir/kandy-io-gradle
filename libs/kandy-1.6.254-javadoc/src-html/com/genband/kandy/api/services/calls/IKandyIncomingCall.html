010package com.genband.kandy.api.services.calls;
011
012/**
013 * This interface is used to handle incoming calls. You can do one of the
014 * following:
015 * <ul>
016 * <li>Accept
017 * <li>Ignore
018 * <li>Reject
019 * </ul>
020 */
021
022public interface IKandyIncomingCall extends IKandyCall
023{
024        /**
025         * Accept the incoming call
026         * 
027         * @param isVideoEnabled
028         *            enables outgoing video for the call
029         * @param listener The callback that will be invoked
030         *            the delegate to get the response
031         */
032        public void accept(boolean isVideoEnabled, KandyCallResponseListener listener);
033
034        /**
035         * Ignore the incoming call
036         * 
037         * @param listener The callback that will be invoked
038         *            the delegate to get the response
039         */
040        public void ignore(KandyCallResponseListener listener);
041
042        /**
043         * Reject the incoming call
044         * 
045         * @param listener The callback that will be invoked
046         *            the delegate to get the response
047         */
048        public void reject(KandyCallResponseListener listener);
049
050}
