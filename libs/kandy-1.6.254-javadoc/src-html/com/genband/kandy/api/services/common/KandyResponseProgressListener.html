010package com.genband.kandy.api.services.common;
011
012import android.net.Uri;
013
014import com.genband.kandy.api.services.chats.IKandyTransferProgress;
015
016public abstract class KandyResponseProgressListener extends KandyBaseResponseListener
017{       
018        
019        /**
020         * Callback has been called after response succeed and pass over parameter the uri of the downloaded file on device
021         * @param fileUri URI of downloaded file
022         */
023        public abstract void onRequestSucceded(Uri fileUri);
024
025        /**
026         * This method is used to display any form of progress in the user interface
027         * while the background computation is still executing. For instance, it can
028         * be used to animate a progress bar or show logs in a text field.
029         * 
030         * @param progress The values indicating progress.
031         */
032        public abstract void onProgressUpdate(IKandyTransferProgress progress);
033
034}
