010package com.genband.kandy.api.services.events;
011
012import java.util.List;
013
014import com.genband.kandy.api.services.calls.KandyRecord;
015import com.genband.kandy.api.services.chats.KandyChatServiceNotificationListener;
016import com.genband.kandy.api.services.common.KandyResponseListener;
017
018/**
019 * This interface is used to handle system events. 
020 * 
021 *
022 */
023 
024
025public interface IKandyEventsService
026{
027        /**
028         * Pull Pending events, Triggers all observers that are registered,
029         * and notify via callback about success or fail 
030         * @param listener notify about operation succeed or failed
031         */
032        public void pullPendingEvents(KandyResponseListener listener);
033
034        /**
035         * Get all event history for KandyRecord from last event timeStamp
036         * @param destination - the destination KandyRecord contact OR group 
037         * @param numberOfEvents - number of messages to return
038         * @param timestamp - last event time stamp
039         * @param moveBackword -  BACKWORD or FORWARD from time stamp
040         * @param listener - notify about operation succeed or failed
041         */
042        public void pullHistoryEvents(KandyRecord destination, int numberOfEvents, 
043                        long timestamp, boolean moveBackword,
044                        KandyPullHistoryEventsListner listener);
045        
046        /**
047         * Get all conversations 
048         * @param listener - notify about operation succeed or failed
049         */
050        public void getAllConversations(final KandyAllConversationsListener listener);
051        
052        /**
053         * pull all conversations with messages from last event timeStamp
054         * @param numberOfEvents - Number of messages to return
055         * @param timestamp - last event time stamp
056         * @param moveBackword -  BACKWORD or FORWARD from time stamp
057         * @param listener - notify about operation succeed or failed
058         */
059        public void pullAllConversationsWithMessages(int numberOfEvents, long timestamp, boolean moveBackword, 
060                        final KandyAllConverstionWithMessagesListener listener);
061        
062        
063        /**
064         * Delete history events - delete events in conversation by user or group conversation  
065         * @param eventUUIDs - events uuid to delete 
066         * @param destination - the user or group that the events is belong to this user or group
067         * @param listener
068         */
069        public void deleteHistoryEvents(List<String> eventUUIDs, KandyRecord destination, KandyResponseListener listener);
070
071        /**
072         * Delete conversations - delete list of conversations 
073         * @param destinations - list of users or groups KandyRecored to delete 
074         * @param listener
075         */
076        public void deleteConversations(List<KandyRecord> destinations, KandyResponseListener listener);
077
078        /**
079         * Delete all conversations
080         * @param listener
081         */
082        public void deleteAllConversations(KandyResponseListener listener);
083
084        
085        /**
086         * Register a callback to be invoked when a service event is received.
087         * 
088         * @param listener The callback that will be invoked
089         */
090        public void registerNotificationListener(KandyEventsServiceNotificationListener listener);
091
092        /** Unregisters - to stop receiving ongoing service events */
093        public void unregisterNotificationListener(KandyEventsServiceNotificationListener listener);
094}
