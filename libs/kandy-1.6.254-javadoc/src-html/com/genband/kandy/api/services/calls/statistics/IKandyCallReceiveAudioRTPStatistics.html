010package com.genband.kandy.api.services.calls.statistics;
011
012/**
013 * Defines statistics of data which was received during the audio call
014 */
015public interface IKandyCallReceiveAudioRTPStatistics extends IKandyCallStatistic {
016        /**
017         * weighted average call delay in msec
018         */
019        public int getAverageLatency();
020
021        /**
022         * weighted average call delay in msec
023         */
024        public int getJitter();
025
026        /**
027         * cumulative packets received since begin of call
028         */
029        public int getReceivedPackets();
030
031        /**
032         * cumulative packet loss since begin of call
033         */
034        public int getReceivedPacketsLost();
035
036        /**
037         * cumulative amount of bytes (audio)
038         */
039        public long getBytesReceived();
040
041        /**
042         * Codec name: Opus , PCMA , PCMU
043         */
044        public String getCodecName();
045
046        // local computed values section
047        
048        /**
049         * Get calculated MOS value for the current quality report
050         * 
051         * @return value between 1 and 5, if  returned <b>-1 </b>
052         * this is indicator of error, and we didn't update the MOS because of 
053         * missing data in SPiDR response
054         */
055        public float getMOS();
056
057        /**
058         * @param avarage
059         *            jitter (computed locally)
060         */
061        int getWeightedAverageJitter();
062
063        /**
064         * @param mAverageMOS
065         */
066        float getAverageMOS();
067
068        int getWeightedAverageLatency();
069
070        int getIntervalReceivedPacketsLost();
071
072        int getIntervalReceivedPackets();
073
074        float getPacketLossPercentage();
075
076}
