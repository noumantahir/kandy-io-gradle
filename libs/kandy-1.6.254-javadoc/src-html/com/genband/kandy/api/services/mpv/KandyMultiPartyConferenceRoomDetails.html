004package com.genband.kandy.api.services.mpv;
005
006import java.util.ArrayList;
007import java.util.List;
008
009import org.json.JSONArray;
010import org.json.JSONException;
011import org.json.JSONObject;
012
013import android.text.TextUtils;
014
015import com.genband.kandy.api.services.mpv.IKandyMultiPartyConferenceParticipant;
016import com.genband.kandy.api.services.mpv.IKandyMultiPartyConferenceRoomDetails;
017import com.genband.kandy.api.services.mpv.KandyMultiPartyConferenceRoom;
018import com.genband.kandy.api.utils.KandyLog;
019import com.genband.kandy.config.JSONKeys.JSONMultiPartiConferenceRoom;
020import com.genband.kandy.core.common.IJSONParse;
021
022public class KandyMultiPartyConferenceRoomDetails implements IKandyMultiPartyConferenceRoomDetails, IJSONParse{
023
024        private static final String TAG = "KandyMultiPartyConferenceRoomDetails";
025        
026        KandyMultiPartyConferenceRoom mConferenceRoom = new KandyMultiPartyConferenceRoom();
027        int mCallDuration;
028        List<IKandyMultiPartyConferenceParticipant> mParticipants = new ArrayList<IKandyMultiPartyConferenceParticipant>();
029        List<String> mAdministrators = new ArrayList<String>();
030
031        
032        @Override
033        public KandyMultiPartyConferenceRoom getConferenceRoom() {
034                return mConferenceRoom;
035        }
036        
037        @Override
038        public int getCallDuration() {
039                return mCallDuration;
040        }
041        
042        @Override
043        public List<IKandyMultiPartyConferenceParticipant> getParticipants() {
044                return mParticipants;
045        }
046        
047        @Override
048        public List<String> getAdministrators() {
049                return mAdministrators;
050        }
051
052        public void setConferenceRoom(KandyMultiPartyConferenceRoom conferenceRoom) {
053                this.mConferenceRoom = conferenceRoom;
054        }
055
056        public void setCallDuration(int callDuration) {
057                this.mCallDuration = callDuration;
058        }
059
060        public void setParticipants(List<IKandyMultiPartyConferenceParticipant> participants) {
061                this.mParticipants = participants;
062        }
063
064        public void setAdministrators(List<String> administrators) {
065                this.mAdministrators = administrators;
066        }
067
068        
069        
070
071        @Override
072        public String toString() {
073                return "KandyMultiPartyConferenceRoomDetails [mConferenceRoom= " + mConferenceRoom + 
074                                ", mCallDuration= " + mCallDuration + 
075                                ", mParticipants= " + mParticipants + 
076                                ", mAdministrators= " + mAdministrators + " ]";
077        }
078        
079        @Override
080        public void initFromJson(JSONObject jsonObject) {
081
082                mConferenceRoom = new KandyMultiPartyConferenceRoom();
083                mConferenceRoom.initFromJson(jsonObject);
084                
085                try
086                {
087                        mCallDuration = jsonObject.getInt(JSONMultiPartiConferenceRoom.KEY_ROOM_DURATION);
088                }
089                catch (Exception e) 
090                {
091                        KandyLog.w(TAG, "initFromJson: " + " " + e.getLocalizedMessage());
092                }
093                
094                mParticipants.clear();
095                try
096                {
097                        JSONArray participantJSONArray = jsonObject.getJSONArray(JSONMultiPartiConferenceRoom.KEY_ROOM_PARTICIPANTS);
098                        for (int i = 0; i < participantJSONArray.length(); i++) {
099                                JSONObject participantJSON = participantJSONArray.getJSONObject(i);
100                                KandyMultiPartyConferenceParticipant participant = new KandyMultiPartyConferenceParticipant();
101                                participant.initFromJson(participantJSON);
102                                mParticipants.add(participant);
103                        }
104                }
105                catch (Exception e) 
106                {
107                        KandyLog.w(TAG, "initFromJson: " + " " + e.getLocalizedMessage());
108                }
109                
110                mAdministrators.clear();
111                try
112                {
113                        JSONArray administratorsJSONArray = jsonObject.getJSONArray(JSONMultiPartiConferenceRoom.KEY_ROOM_ADMINISTRATORS);
114                        for (int i = 0; i < administratorsJSONArray.length(); i++) {
115                                JSONObject adminsJSON = administratorsJSONArray.getJSONObject(i);
116                                String adminId = adminsJSON.getString(JSONMultiPartiConferenceRoom.KEY_ROOM_PARTICIPANT_ID);
117                                if (TextUtils.isEmpty(adminId) == false) {
118                                        mAdministrators.add(adminId);
119                                }
120
121                        }
122                }
123                catch (Exception e) 
124                {
125                        KandyLog.w(TAG, "initFromJson: " + " " + e.getLocalizedMessage());
126                }
127                
128        }
129
130        @Override
131        public JSONObject toJson() {
132        
133                JSONObject roomWrapper = mConferenceRoom.toJson();
134
135                try 
136                {
137                        roomWrapper.put(JSONMultiPartiConferenceRoom.KEY_ROOM_DURATION, mCallDuration);
138                }
139                catch (JSONException e) 
140                {
141                        KandyLog.w(TAG, "toJson: " + " " + e.getLocalizedMessage());
142                }
143                
144                try {
145                        JSONArray participantsJsonArray = new JSONArray();
146                        for (int i = 0; i < mParticipants.size(); i++) 
147                        {
148                                KandyMultiPartyConferenceParticipant kandyParticipant = (KandyMultiPartyConferenceParticipant) mParticipants.get(i);
149                                participantsJsonArray.put(kandyParticipant.toJson());
150                        }
151
152                        roomWrapper.put(JSONMultiPartiConferenceRoom.KEY_ROOM_PARTICIPANTS, participantsJsonArray);
153                } 
154                catch (JSONException e) 
155                {
156                        KandyLog.w(TAG, "toJson: " + " " + e.getLocalizedMessage());
157                }
158
159                
160                try {
161                        JSONArray adminJsonArray = new JSONArray();
162                        for (int i = 0; i < mAdministrators.size(); i++) 
163                        {
164                                String admin = mAdministrators.get(i);
165                                JSONObject adminsJSON = new JSONObject();
166                                adminsJSON.put(JSONMultiPartiConferenceRoom.KEY_ROOM_PARTICIPANT_ID, admin);
167                                adminJsonArray.put(adminsJSON);
168                        }
169                        roomWrapper.put(JSONMultiPartiConferenceRoom.KEY_ROOM_ADMINISTRATORS, adminJsonArray);
170                } 
171                catch (JSONException e) 
172                {
173                        KandyLog.w(TAG, "toJson: " + " " + e.getLocalizedMessage());
174                }
175                
176                return roomWrapper;
177        }
178        
179}
