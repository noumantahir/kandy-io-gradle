059        /**
060         * Get the average latency of the call
061         */
062        public void update(CallStatistic[] callStatistics) {
063                int capacity = 0;
064                
065                //For the large strings it is hard to allocate memory dynamically 
066                for (CallStatistic stat : callStatistics)
067                {
068                        capacity = capacity + stat.toString().length();
069                }
070                KandyLog.d(TAG, "update: " + " statistics length: " + capacity);
071                StringBuilder builder = new StringBuilder(capacity);
072                for (CallStatistic stat : callStatistics) {
073                        builder.append(stat.toString());
074                        KandyCallStatisticHolder kandyCallStatistic = new KandyCallStatisticHolder(stat.id, stat.timestamp, stat.type, stat.values);
075                        mStatisticsList.add(kandyCallStatistic);
076                        IKandyCallStatistic statistic = CallStatisticParser.getStatistic(kandyCallStatistic);
077                        if (statistic instanceof KandyCallGeneralAudioRTPStatistics) {
078                                mKandyCallGeneralAudioRTPStatistics = (KandyCallGeneralAudioRTPStatistics) statistic;
079                        } else if (statistic instanceof IKandyCallGeneralVideoRTPStatistics) {
080                                mKandyCallGeneralVideoRTPStatistics = (KandyCallGeneralVideoRTPStatistics) statistic;
081                        } else if (statistic instanceof KandyCallReceiveAudioRTPStatistics) {
082                                int packetsReceivedOld = 0;
083                                int packetsReceivedLostOld = 0;
084                                int weightedAvarageJitter = 0;
085                                int weightedAvarageLatency = 0;
086                                float averageMos = 0;
087                                if(mKandyCallReceiveAudioRTPStatistics!=null){
088                                        packetsReceivedOld = mKandyCallReceiveAudioRTPStatistics.getReceivedPackets();
089                                        packetsReceivedLostOld = mKandyCallReceiveAudioRTPStatistics.getReceivedPacketsLost();
090                                        weightedAvarageJitter = mKandyCallReceiveAudioRTPStatistics.getWeightedAverageJitter();
091                                        weightedAvarageLatency = mKandyCallReceiveAudioRTPStatistics.getWeightedAverageLatency();
092                                        averageMos = mKandyCallReceiveAudioRTPStatistics.getAverageMOS();
093                                }
094                                mKandyCallReceiveAudioRTPStatistics = (KandyCallReceiveAudioRTPStatistics) statistic;
095                                int packetsReceivedNew = mKandyCallReceiveAudioRTPStatistics.getReceivedPackets();
096                                int packetsReceivedLostNew = mKandyCallReceiveAudioRTPStatistics.getReceivedPacketsLost();
097                                mKandyCallReceiveAudioRTPStatistics.setIntervalReceivedPackets((packetsReceivedNew - packetsReceivedOld));
098                                mKandyCallReceiveAudioRTPStatistics.setIntervalReceivedPacketsLost((packetsReceivedLostNew - packetsReceivedLostOld));
099                                mKandyCallReceiveAudioRTPStatistics.setWeightedAverageJitter(weightedAvarageJitter);
100                                mKandyCallReceiveAudioRTPStatistics.setWeightedAverageLatency(weightedAvarageLatency);
101                                mKandyCallReceiveAudioRTPStatistics.setAverageMOS(averageMos);
102                        } else if (statistic instanceof KandyCallReceiveVideoRTPStatistics) {
103                                mKandyCallReceiveVideoRTPStatistics = (KandyCallReceiveVideoRTPStatistics) statistic;
104                        } else if (statistic instanceof KandyCallSendAudioRTPStatistics) {
105                                mKandyCallSendAudioRTPStatistics = (KandyCallSendAudioRTPStatistics) statistic;
106                        } else if (statistic instanceof KandyCallSendVideoRTPStatistics) {
107                                mKandyCallSendVideoRTPStatistics = (KandyCallSendVideoRTPStatistics) statistic;
108                        }
109                }
110                this.mRawData = builder.toString();
111                KandyLog.d(TAG, "update: new Raw Data: [" + " " + this.mRawData +"]" );
112                
113                if (mKandyCallReceiveAudioRTPStatistics.getIntervalReceivedPackets() > 0)
114                {
115                        updateMOS();
116                }
117                updateAverages();
118        }
119
120        private void updateAverages() {
121                int jitter = mKandyCallReceiveAudioRTPStatistics.getJitter();
122                int averageLatency = mKandyCallReceiveAudioRTPStatistics.getAverageLatency();
123                mKandyCallReceiveAudioRTPStatistics.setWeightedAverageJitter((int) (mKandyCallReceiveAudioRTPStatistics.getWeightedAverageJitter() == 0 ? jitter
124                                : mKandyCallReceiveAudioRTPStatistics.getWeightedAverageJitter() * HISTORY_WEIGHT + jitter * CURRENT_WEIGHT));
125                mKandyCallReceiveAudioRTPStatistics.setWeightedAverageLatency((int) (mKandyCallReceiveAudioRTPStatistics.getWeightedAverageLatency() == 0 ? averageLatency
126                                : mKandyCallReceiveAudioRTPStatistics.getWeightedAverageLatency() * HISTORY_WEIGHT + averageLatency * CURRENT_WEIGHT));
127                
128                if(mKandyCallReceiveAudioRTPStatistics.getMOS() < 0) {
129                        /* We start the MOS from -1 if we continue to receive -1 it's mean that we didn't update the MOS,
130                         * because didn't get necessary data from SPiDR, so we wouldn't update average MOS because it would be invalid */
131                        KandyLog.d(TAG, "updateAverages: " + " MOS is < 0, so average MOS wouldn't be updated");
132                        
133                        return;
134                }
135                
136                if (mKandyCallReceiveAudioRTPStatistics.getAverageMOS() == 0f) {
137                        mKandyCallReceiveAudioRTPStatistics.setAverageMOS(mKandyCallReceiveAudioRTPStatistics.getMOS());
138                } else {
139                        mKandyCallReceiveAudioRTPStatistics.setAverageMOS((float) (mKandyCallReceiveAudioRTPStatistics.getAverageMOS() * HISTORY_WEIGHT + mKandyCallReceiveAudioRTPStatistics.getMOS() * CURRENT_WEIGHT));
140                }
141        }
142
143        private void updateMOS() {
144                // mPacketReceived cann't be 0 because it is divider
145                if (mKandyCallReceiveAudioRTPStatistics.getIntervalReceivedPackets() == 0 && mKandyCallReceiveAudioRTPStatistics.getIntervalReceivedPacketsLost() == 0) {
146                        KandyLog.d(TAG, "updateMOS: " + " no packeges received or lost no reason to calculate");
147                        return;
148                } else if (mKandyCallReceiveAudioRTPStatistics.getIntervalReceivedPackets() == 0) {
149                        KandyLog.d(TAG, "updateMOS: " + " no packeges received no reason to calculate");
150                        return;
151                } else {
152            // mPacketLossPercentage value must be between 0..100
153            mKandyCallReceiveAudioRTPStatistics.setPacketLossPercentage(((float) (100.0 * ((float)mKandyCallReceiveAudioRTPStatistics.getIntervalReceivedPacketsLost() / (float)mKandyCallReceiveAudioRTPStatistics.getIntervalReceivedPackets()))));
154                }
155
156                calcMOS();
157
158        }
159
160        private void calcMOS() {
161                float RVal = calcRValue();
162
163                // Convert the R into an MOS value. (this is a known formula)
164                float mos = (float) (1 + (0.035) * RVal + (0.000007) * RVal * (RVal - 60) * (100 - RVal));
165
166                if (mos > 5) {
167                        KandyLog.d(TAG, "calcMOS: " + "Problematic MOS, changing it to 1. OriginalMOS - " + mos
168                                        + ", RValue - " + RVal + ", packetLossPercentage - " + String.format("%3f", mKandyCallReceiveAudioRTPStatistics.getPacketLossPercentage())
169                                        + ", avarageLatency - " + mKandyCallReceiveAudioRTPStatistics.getAverageLatency() + ", jitter - " + mKandyCallReceiveAudioRTPStatistics.getJitter());
170                        mos = 1;
171                }
172                mKandyCallReceiveAudioRTPStatistics.setMOS(mos);
173        }
174
175        private float calcRValue() {
176                synchronized (lockObj) {
177                        float RVal = 0f;
178                        // Take the average latency, add jitter, but double the impact to
179                        // latency
180                        // then add 10 for protocol latencies
181                        
182                        long jitter = mKandyCallReceiveAudioRTPStatistics.getJitter();
183                        long averageLatency = mKandyCallReceiveAudioRTPStatistics.getAverageLatency();
184                        float effectiveLatency = (float) (averageLatency + jitter);
185
186                        // Implement a basic curve - deduct 4 for the R value at 160ms of
187                        // latency
188                        // (round trip). Anything over that gets a much more agressive
189                        // deduction
190                        if (effectiveLatency < 300) {
191                                RVal = (float) (93.2 - (effectiveLatency / 40));
192                        } else {
193                                RVal = (float) (93.2 - (effectiveLatency - 150) / 20);
194                        }
195
196                        if (RVal < 0f) {
197                                RVal = 0f;
198                        }
199            else {
200                // Now, let's deduct 25 R values per percentage of packet loss
201                RVal = (float) (RVal - (mKandyCallReceiveAudioRTPStatistics.getPacketLossPercentage() * 2.5));
202                //Need to make sure we don't get minus values: After we reduced EffectiveLatency need to do follow:
203                if (RVal < 0f) {
204                    RVal = 0f;
205                }
206            }
207
208            return RVal;
209                }
210
211        }

218        @Override
219        public JSONObject toJson() {
220
221                JSONObject audioSentJson = mKandyCallSendAudioRTPStatistics.toJson();
222                JSONObject audioReceivedJson = mKandyCallReceiveAudioRTPStatistics.toJson();
223                JSONObject audioGeneralJson = mKandyCallGeneralAudioRTPStatistics.toJson();
224                JSONObject videoSentJson = mKandyCallSendVideoRTPStatistics.toJson();
225                JSONObject videoReceivedJson = mKandyCallReceiveVideoRTPStatistics.toJson();
226                JSONObject videoGeneralJson = mKandyCallGeneralVideoRTPStatistics.toJson();
227                JSONObject json = combineJsonObjects(audioSentJson, audioReceivedJson, audioGeneralJson, videoSentJson, videoReceivedJson, videoGeneralJson);
228                return json;
229        }
230        
231        private JSONObject combineJsonObjects(JSONObject... jsons){
232                JSONObject combinedJson = new JSONObject();
233                for(int i=0; i<jsons.length; i++) {
234                        JSONObject json = jsons[i];
235                        Iterator<?> keys = json.keys();
236                        while( keys.hasNext() ) {
237                            String key = (String)keys.next();
238                            try {
239                                        combinedJson.putOpt(key, json.get(key));
240                                } catch (JSONException e) {
241                                        KandyLog.e(TAG, "combineJsonObjects: " + " " + e.getLocalizedMessage());
242                                }
243                        }
244                }
245                return combinedJson;
246        }

250                JSONObject audioJson = json.optJSONObject(JSONKeys.KEY_REPORT_AUDIO_SECTION);
251                if (audioJson != null) {
252                        JSONObject audioSentJson = audioJson.optJSONObject(JSONKeys.KEY_REPORT_SENT_SECTION);
253                        if (mKandyCallSendAudioRTPStatistics == null) {
254                                mKandyCallSendAudioRTPStatistics = new KandyCallSendAudioRTPStatistics();
255                        }
256                        if (audioSentJson != null) {
257                                mKandyCallSendAudioRTPStatistics.initFromJson(audioSentJson);
258                        }
259
260                        JSONObject audioReceiveJson = audioJson.optJSONObject(JSONKeys.KEY_REPORT_RECEIVE_SECTION);
261                        if (mKandyCallReceiveAudioRTPStatistics == null) {
262                                mKandyCallReceiveAudioRTPStatistics = new KandyCallReceiveAudioRTPStatistics();
263                        }
264                        if (audioReceiveJson != null) {
265                                mKandyCallReceiveAudioRTPStatistics.initFromJson(audioReceiveJson);
266                        }
267
268                        JSONObject audioGeneralJson = audioJson.optJSONObject(JSONKeys.KEY_REPORT_GENERAL_SECTION);
269                        if (mKandyCallGeneralAudioRTPStatistics == null) {
270                                mKandyCallGeneralAudioRTPStatistics = new KandyCallGeneralAudioRTPStatistics();
271                        }
272                        if (audioGeneralJson != null) {
273                                mKandyCallGeneralAudioRTPStatistics.initFromJson(audioGeneralJson);
274                        }
275                }
276
277                JSONObject videoJson = json.optJSONObject(JSONKeys.KEY_REPORT_VIDEO_SECTION);
278                if (videoJson != null) {
279                        JSONObject videoSentJson = videoJson.optJSONObject(JSONKeys.KEY_REPORT_SENT_SECTION);

281                                mKandyCallSendVideoRTPStatistics = new KandyCallSendVideoRTPStatistics();
282                        }
283                        if (videoSentJson != null) {
284                                mKandyCallSendVideoRTPStatistics.initFromJson(videoSentJson);
285                        }
286
287                        JSONObject videoReceiveJson = videoJson.optJSONObject(JSONKeys.KEY_REPORT_RECEIVE_SECTION);
288                        if (mKandyCallReceiveVideoRTPStatistics == null) {
289                                mKandyCallReceiveVideoRTPStatistics = new KandyCallReceiveVideoRTPStatistics();
290                        }
291                        if (videoReceiveJson != null) {
292                                mKandyCallReceiveVideoRTPStatistics.initFromJson(videoReceiveJson);
293                        }
294
295                        JSONObject videoGeneralJson = videoJson.optJSONObject(JSONKeys.KEY_REPORT_GENERAL_SECTION);
296                        if (mKandyCallGeneralVideoRTPStatistics == null) {
297                                mKandyCallGeneralVideoRTPStatistics = new KandyCallGeneralVideoRTPStatistics();
298                        }
299                        if (videoGeneralJson != null) {

302                }
