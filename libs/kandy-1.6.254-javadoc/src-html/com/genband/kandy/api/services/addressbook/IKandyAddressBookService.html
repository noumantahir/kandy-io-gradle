010package com.genband.kandy.api.services.addressbook;
011
012import com.genband.kandy.api.services.common.KandyResponseListener;
013
014
015/**
016 * @author Michaell
017 *
018 */
019public interface IKandyAddressBookService
020{
021
022        /**
023         * register to ongoing notification
024         * 
025         * @param listener
026         *            The callback that will be invoked
027         */
028        public void registerNotificationListener(KandyAddressBookServiceNotificationListener listener);
029
030        /**
031         * unregister from ongoing notification
032         * 
033         * @param listener
034         *            The callback that will be invoked
035         */
036        public void unregisterNotificationListener(KandyAddressBookServiceNotificationListener listener);
037        
038        /**
039         * Get device contacts.
040         * <p>
041         * 
042         * <b>Note:</b> This API requires android.permission.READ_CONTACTS
043         * 
044         * @param filters
045         *            The filter of which contacts to return. Passing null will
046         *            return all visible contacts that has phone number
047         * @param listner
048         *            The callback that will be invoked
049         * 
050         */
051        public void getDeviceContacts(KandyDeviceContactsFilter[] filters, KandyContactsListener listner);
052
053        /**
054         *  Returns the domain contacts as {@link IKandyContact} objects in an array in the response callback. If operation failed, return an error.
055         *
056         *  @param listener a callback with {@link IKandyContact} objects in an array if operation succeeded or error if operation failed.
057         */
058        public void getDomainDirectoryContacts(KandyContactsListener listener);
059        
060        /**
061         *  Returns the domain contacts as {@link IKandyContact} objects in an array in the response callback. If operation failed, return an error. Filters contact according to filterType and text in textSearch.
062         *
063         *  @param filter       the fields in which to search for the text.
064         *  @param isCaseSensitive    Boolean value indicating whether or not the search is case sensitive.
065         *  @param search_string       the text to search in the fields indicated by filterType
066         *  @param listener A callback with {@link IKandyContact} objects in an array if operation succeeded, or error if operation failed.
067         */
068        public void getFilteredDomainDirectoryContacts(KandyDomainContactFilter filter, boolean isCaseSensitive, String search_string, KandyContactsListener listener);
069
070        /**
071         * Returns the user personal address book {@link IKandyContact}  in an array in the response callback. If operation failed, return an error.
072         * @param listener A callback with {@link IKandyContact} objects in an array if operation succeeded, or error if operation failed.
073         */
074        public void getPersonalAddressBook(KandyContactsListener listener);
075        
076        /**
077         * Returns the added user server id
078         * @param contactParams {@link KandyContactParams} user to add
079         * @param listener A callback with {@link String} server contact id if operation succeeded, or error if operation failed
080         */
081        public void addContactToPersonalAddressBook(KandyContactParams contactParams, KandyContactListener listener);
082        
083        /**
084         * Removes contact from address book
085         * @param contactId {@link String} contact id of the user to delete
086         * @param listener The callback that will be invoked
087         */
088        public void removePersonalAddressBookContact(String contactId, KandyResponseListener listener);
089        
090}
