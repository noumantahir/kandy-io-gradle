001package com.genband.kandy.api.services.calls.statistics;
002
003import org.webrtc.StatsReport.Value;
004
005/**
006 * Contains data of the current call statistic's item
007 */
008public class KandyCallStatisticHolder {
009        
010        String id;
011        double timestamp;
012        String type;
013        Value[] values;
014        
015        public KandyCallStatisticHolder(String id, double timestamp, String type, Value[] values) {
016                
017                this.id = id;
018                this.timestamp = timestamp;
019                this.type = type;
020                this.values = values;
021        }
022        
023        /**
024         * Get statistic's id
025         */
026        public String getId() {
027                return this.id;
028        }
029        
030        /**
031         * Get statistic's timestamp(in millis)
032         */
033        public double getTimestamp() {
034                return this.timestamp;
035        }
036
037        /**
038         * Get statistic's type
039         * @return
040         */
041        public String getType() {
042                return this.type;
043        }
044        
045        /**
046         * Get statistic's values
047         */
048        public Value[] getValues() {
049                return this.values;
050        }
051}
