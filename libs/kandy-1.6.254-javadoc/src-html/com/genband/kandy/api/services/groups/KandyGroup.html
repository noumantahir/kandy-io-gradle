010package com.genband.kandy.api.services.groups;
011
012import com.genband.kandy.api.Kandy;
013import com.genband.kandy.api.services.calls.KandyRecord;
014import com.genband.kandy.api.services.chats.IKandyImageItem;
015
016import java.util.Date;
017import java.util.List;
018
019/**
020 * Contains the group's data
021 *
022 */
023public class KandyGroup implements IKandyGroup {
024
025        private List<KandyGroupParticipant> mGroupParticipants;
026
027        private KandyRecord mGroupId;
028
029        private IKandyImageItem mGroupImageData;
030
031        private String mGroupName;
032
033        private Date mGroupCreationDate;
034
035        private boolean isGroupMuted;
036
037        private int mGroupMaxParticipants;
038
039        public KandyGroup()
040        {
041        }
042
043        /**
044         * Constructor for KandyGroup with name
045         * @param groupName name of the group
046         */
047        public KandyGroup(String groupName) {
048                this();
049                mGroupName = groupName;
050        }
051
052        /**
053         * Constructor for KandyGroup with name and image
054         * @param groupName name of the group
055         * @param imageItem image of the group
056         */
057        public KandyGroup(String groupName, IKandyImageItem imageItem) {
058                this(groupName);
059                mGroupImageData = imageItem;
060        }
061
062        /**
063         * Set id for group
064         * @param groupId to be set to the current group
065         */
066        public void setGroupId(KandyRecord groupId) {
067                mGroupId = groupId;
068        }
069
070        /**
071         * Set maximal number of group's participants
072         * @param members
073         */
074        public void setMaxMembers(int members) {
075                mGroupMaxParticipants = members;
076        }
077
078        /**
079         * Set grpoup's creation date
080         * @param date to be set
081         */
082        public void setCreationDate(Date date) {
083                mGroupCreationDate = date;
084        }
085        
086        /**
087         * Sets the name of the group
088         * @param groupName name of the group
089         */
090        public void setGroupName(String groupName) {
091                mGroupName = groupName;
092        }
093        
094        /**
095         * Sets the image of the group
096         * @param imageItem image to be set to the group
097         */
098        public void setGroupImage(IKandyImageItem imageItem) {
099                mGroupImageData = imageItem;
100        }
101        
102        /**
103         * Add the participants to the group
104         * @param groupParticipants participants to be added to the group
105         */
106        public void setGroupParticipants(List<KandyGroupParticipant> groupParticipants)
107        {
108                this.mGroupParticipants = groupParticipants;
109        }
110
111        /**
112         * Set group as muted
113         * @param isMuted true if mute otherwise unmute
114         */
115        public void setGroupMute(boolean isMuted) {
116                this.isGroupMuted = isMuted;
117        }
118
119        /**
120         * Set member as muted
121         * @param user to be set as muted
122         * @param isMuted true if mute otherwise unmute
123         */
124        public void setParticipantMute(KandyRecord user, boolean isMuted) {
125                for(int i=0; i<mGroupParticipants.size(); i++) {
126                        if(mGroupParticipants.get(i).getParticipant().equals(user)) {
127                                mGroupParticipants.get(i).setMute(isMuted);
128                                return;
129                        }
130                }
131        }
132
133        @Override
134        public KandyRecord getGroupId() {
135                return mGroupId;
136        }
137
138        @Override
139        public int getMaxParticipantsNumber() {
140                return mGroupMaxParticipants;
141        }
142
143        @Override
144        public Date getCreationDate() {
145                return mGroupCreationDate;
146        }
147
148        @Override
149        public boolean isGroupMuted() {
150                return isGroupMuted;
151        }
152
153        @Override
154        public boolean isParticipantMuted(KandyRecord participant) {
155                for(KandyGroupParticipant member : mGroupParticipants) {
156                        if(member.equals(participant)) {
157                                return true;
158                        }
159                }
160
161                return false;
162        }
163
164        @Override
165        public String getGroupName() {
166                return mGroupName;
167        }
168
169        @Override
170        public IKandyImageItem getImage() {
171                return mGroupImageData;
172        }
173
174        @Override
175        public List<KandyGroupParticipant> getGroupParticipants()
176        {
177                return mGroupParticipants;
178        }
179
180        @Override
181        public KandyGroupParticipant getSelfParticipant()
182        {
183                KandyGroupParticipant me = null;
184                List<KandyGroupParticipant> groupParticipants = getGroupParticipants();
185                if (groupParticipants != null)
186                {
187                        String self = Kandy.getSession().getKandyUser().getUserId();
188                        if (self != null)
189                        {
190                                for (KandyGroupParticipant participant : groupParticipants)
191                                {
192                                        if (self.equals(participant.getParticipant().getUri()))
193                                        {
194                                                me = participant;
195                                        }
196                                }
197                        }
198                }
199
200                return me;
201        }
202
203        @Override
204        public String toString()
205        {
206                return "KandyGroup [mGroupParticipants=" + mGroupParticipants + ", mGroupId=" + mGroupId + ", mGroupImageData="
207                                + mGroupImageData + ", mGroupName=" + mGroupName + ", mGroupCreationDate=" + mGroupCreationDate
208                                + ", isGroupMuted=" + isGroupMuted + ", mGroupMaxParticipants=" + mGroupMaxParticipants + "]";
209        }
210
211}
